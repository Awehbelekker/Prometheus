## 💻 COMPLETE USAGE EXAMPLE

```python
import asyncio
from core.hrm_integration import HRMTradingEngine
from core.reasoning.thinkmesh_enhanced import EnhancedThinkMeshAdapter, ThinkMeshConfig, ReasoningStrategy
from core.multimodal_analyzer import analyze_trading_chart
from core.ensemble_voting_system import ensemble_trading_decision
from core.universal_reasoning_engine import UniversalReasoningEngine

async def ultimate_trading_decision(symbol, chart_path):
    """
    Complete AI-powered trading decision using ALL systems
    """
    
    # 1. HRM Hierarchical Reasoning
    hrm_engine = HRMTradingEngine()
    hrm_result = await hrm_engine.reason_trading_decision({
        'symbol': symbol,
        'action': 'analyze'
    })
    
    print(f"HRM Analysis: {hrm_result}")
    
    # 2. Visual Chart Analysis (Multimodal)
    chart_result = analyze_trading_chart(chart_path, symbol, "1D")
    
    print(f"Chart Patterns: {chart_result.patterns_detected}")
    print(f"Trend: {chart_result.trend_direction}")
    
    # 3. ThinkMesh with Official DeepConf
    thinkmesh = EnhancedThinkMeshAdapter(enabled=True)
    config = ThinkMeshConfig(
        strategy=ReasoningStrategy.DEEPCONF,  # Official DeepConf!
        model_name="deepseek-r1:8b"
    )
    
    question = f"Should I buy {symbol} given the {chart_result.trend_direction} trend?"
    
    thinkmesh_result = await thinkmesh.reason(question, config)
    
    print(f"ThinkMesh Confidence: {thinkmesh_result.answer}")
    
    # 4. Ensemble Voting (Multi-model consensus)
    ensemble_result = await ensemble_trading_decision(
        question=question,
        market_data={
            'symbol': symbol,
            'trend': chart_result.trend_direction,
            'patterns': chart_result.patterns_detected,
            'hrm_score': hrm_result.get('confidence', 0.5)
        }
    )
    
    print(f"Ensemble Consensus: {ensemble_result.consensus_confidence:.2f}")
    print(f"Agreement Rate: {ensemble_result.agreement_rate:.1%}")
    
    # 5. Universal Reasoning Engine (Final synthesis)
    reasoning_engine = UniversalReasoningEngine()
    
    final_decision = reasoning_engine.synthesize_decision({
        'hrm': hrm_result,
        'multimodal': {
            'patterns': chart_result.patterns_detected,
            'trend': chart_result.trend_direction,
            'confidence': chart_result.confidence
        },
        'thinkmesh': {
            'answer': thinkmesh_result.answer,
            'strategy': 'deepconf'
        },
        'ensemble': {
            'decision': ensemble_result.final_answer,
            'consensus': ensemble_result.consensus_confidence,
            'agreement': ensemble_result.agreement_rate
        }
    })
    
    print("\n" + "="*80)
    print("FINAL DECISION (ALL SYSTEMS)")
    print("="*80)
    print(f"Decision: {final_decision['action']}")
    print(f"Confidence: {final_decision['confidence']:.2f}")
    print(f"Reasoning: {final_decision['reasoning']}")
    print()
    
    # Use confidence for position sizing
    if final_decision['confidence'] > 0.8:
        position_size = 0.10  # Full position
    elif final_decision['confidence'] > 0.6:
        position_size = 0.05  # Half position
    else:
        position_size = 0.0   # Skip trade
    
    print(f"Position Size: {position_size:.1%}")
    
    return final_decision

# Run complete analysis
asyncio.run(ultimate_trading_decision("AAPL", "charts/aapl.png"))
```
